START

DEFINE SERVER_HOST = "localhost"
DEFINE SERVER_PORT = "9988"
DEFINE SERVER_TYPE = "tcp"

FUNCTION main:
    TRY TO CONNECT TO SERVER (SERVER_TYPE, SERVER_HOST, SERVER_PORT)
    IF CONNECTION FAILS:
        PRINT "Error: Unable to connect to server"
        EXIT PROGRAM

    SEND MESSAGE TO SERVER: "Hello Server! Greetings."

    CREATE BUFFER TO HOLD RESPONSE (1024 BYTES)

    READ RESPONSE FROM SERVER INTO BUFFER
    IF READING FAILS:
        PRINT "Error reading from server"
    ELSE:
        CONVERT BUFFER CONTENTS TO STRING
        PRINT "Received: " + RESPONSE

    CLOSE CONNECTION TO SERVER

END
